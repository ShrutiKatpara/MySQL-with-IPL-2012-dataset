select wicket_taken.player_out, player.player_name, wicket_taken.kind_out
from wicket_taken
natural join player where player.player_id = wicket_taken.player_out and player.player_id not in 
(select wicket_taken.player_out from wicket_taken 
where wicket_taken.kind_out = "caught")
group by wicket_taken.player_out
;

select player_out, kind_out
from wicket_taken
group by player_out, kind_out
;


select * 
from v
where frequency in
(select *
from v
where kind_out = caught
order by frequency)
;

select * 
from oneb
where frequency =  ( select min(frequency) from oneb )
;

drop view oneb;
